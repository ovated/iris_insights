def fibonacci_sequence(n):
    """Fibonacci sequence up to n-th number"""
    # Ensure n is not less than or equal to 0, or equal to 1
    if n <= 0:
        return []
    elif n == 1:
        return [0]

    sequence = [0,1]
    # Compute the fibonacci sequence
    for i in range(2,n):
        sequence.append(sequence[-1] + sequence[-2])
    return sequence
